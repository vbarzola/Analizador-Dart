// Ejemplo de como encontrar el n-esimo 
// numero de fibonacci
// Este ejemplo tiene errores sintacticos
int fibonacci(int n) {
  if (n == 0 || n == 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}
